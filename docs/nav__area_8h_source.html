626 class ShortestPathCost
627 {
628 public:
629  float operator() ( CNavArea *area, CNavArea *fromArea, const CNavLadder *ladder )
630  {
631  if (fromArea == NULL)
632  {
633  // first area in path, no cost
634  return 0.0f;
635  }
636  else
637  {
638  // compute distance travelled along path so far
639  float dist;
640 
641  if (ladder)
642  dist = ladder->m_length;
643  else
644  dist = (*area->GetCenter() - *fromArea->GetCenter()).Length();
645 
646  float cost = dist + fromArea->GetCostSoFar();
647 
648  // if this is a "crouch" area, add penalty
649  if (area->GetAttributes() & NAV_CROUCH)
650  {
651  const float crouchPenalty = 20.0f; // 10
652  cost += crouchPenalty * dist;
653  }
654 
655  // if this is a "jump" area, add penalty
656  if (area->GetAttributes() & NAV_JUMP)
657  {
658  const float jumpPenalty = 5.0f;
659  cost += jumpPenalty * dist;
660  }
661 
662  return cost;
663  }
664  }
665 };

678 template< typename CostFunctor >
679 bool NavAreaBuildPath( CNavArea *startArea, CNavArea *goalArea, const Vector *goalPos, CostFunctor &costFunc, CNavArea **closestArea = NULL )
680 {
681  if (closestArea)
682  *closestArea = NULL;
683 
684  if (startArea == NULL)
685  return false;
686 
687  //
688  // If goalArea is NULL, this function will return the closest area to the goal.
689  // However, if there is also no goal, we can't do anything.
690  //
691  if (goalArea == NULL && goalPos == NULL)
692  {
693  return false;
694  }
695 
696  startArea->SetParent( NULL );
697 
698  // if we are already in the goal area, build trivial path
699  if (startArea == goalArea)
700  {
701  goalArea->SetParent( NULL );
702 
703  if (closestArea)
704  *closestArea = goalArea;
705 
706  return true;
707  }
708 
709  // determine actual goal position
710  Vector actualGoalPos = (goalPos) ? *goalPos : *goalArea->GetCenter();
711 
712  // start search
713  CNavArea::ClearSearchLists();
714 
715  // compute estimate of path length
717  startArea->SetTotalCost( (*startArea->GetCenter() - actualGoalPos).Length() );
718 
719  float initCost = costFunc( startArea, NULL, NULL );
720  if (initCost < 0.0f)
721  return false;
722  startArea->SetCostSoFar( initCost );
723 
724  startArea->AddToOpenList();
725 
726  // keep track of the area we visit that is closest to the goal
727  if (closestArea)
728  *closestArea = startArea;
729  float closestAreaDist = startArea->GetTotalCost();
730 
731  // do A* search
732  while( !CNavArea::IsOpenListEmpty() )
733  {
734  // get next area to check
735  CNavArea *area = CNavArea::PopOpenList();
736 
737  // check if we have found the goal area
738  if (area == goalArea)
739  {
740  if (closestArea)
741  *closestArea = goalArea;
742 
743  return true;
744  }
745 
746  // search adjacent areas
747  bool searchFloor = true;
748  int dir = NORTH;
749  const NavConnectList *floorList = area->GetAdjacentList( NORTH );
750  NavConnectList::const_iterator floorIter = floorList->begin();
751 
752  bool ladderUp = true;
753  const NavLadderList *ladderList = NULL;
754  NavLadderList::const_iterator ladderIter;
755  enum { AHEAD = 0, LEFT, RIGHT, BEHIND, NUM_TOP_DIRECTIONS };
756  int ladderTopDir;
757 
758  while(true)
759  {
760  CNavArea *newArea;
761  NavTraverseType how;
762  const CNavLadder *ladder = NULL;
763 
764  //
765  // Get next adjacent area - either on floor or via ladder
766  //

780  ladderIter = ladderList->begin();

782  }

795  ++floorIter;

813  continue;
814  }

819

821  if (ladder->m_isDangling)
822  {
823  ++ladderIter;
824  continue;
825  }
826 
827  // do not use BEHIND connection, as its very hard to get to when going up a ladder
828  if (ladderTopDir == AHEAD)

849  }

852  continue;
853  }

856  if (newArea == area)
857  continue;

860 
861  // check if cost functor says this area is a dead-end
862  if (newCostSoFar < 0.0f)
863  continue;
864 
865  if ((newArea->IsOpen() || newArea->IsClosed()) && newArea->GetCostSoFar() <= newCostSoFar)
866  {
867  // this is a worse path - skip it
868  continue;
869  }
870  else
871  {
872  // compute estimate of distance left to go
873  float newCostRemaining = (*newArea->GetCenter() - actualGoalPos).Length();
874 
875  // track closest area to goal in case path fails
876  if (closestArea && newCostRemaining < closestAreaDist)
877  {
878  *closestArea = newArea;
879  closestAreaDist = newCostRemaining;
880  }
881 
882  newArea->SetParent( area, how );
883  newArea->SetCostSoFar( newCostSoFar );
884  newArea->SetTotalCost( newCostSoFar + newCostRemaining );
885 
886  if (newArea->IsClosed())
887  newArea->RemoveFromClosedList();
888 
889  if (newArea->IsOpen())
890  {
891  // area already on open list, update the list order to keep costs sorted
892  newArea->UpdateOnOpenList();
893  }
894  else
895  {
896  newArea->AddToOpenList();
897  }
898  }
899  }
900 
901  // we have searched this area
902  area->AddToClosedList();
903  }
904 
905  return false;
906 }
