60  bool Compute( const Vector *start, const Vector *goal, CostFunctor &costFunc )
61  {
62  Invalidate();
63 
64  if (start == NULL || goal == NULL)
65  return false;
66 
67  CNavArea *startArea = TheNavAreaGrid.GetNearestNavArea( start );
68  if (startArea == NULL)
69  return false;
70 
71  CNavArea *goalArea = TheNavAreaGrid.GetNavArea( goal );
72 
73  // if we are already in the goal area, build trivial path
74  if (startArea == goalArea)
75  {
76  BuildTrivialPath( start, goal );
77  return true;
78  }
79 
80  // make sure path end position is on the ground
81  Vector pathEndPosition = *goal;
82  if (goalArea)
83  pathEndPosition.z = goalArea->GetZ( &pathEndPosition );
84  else
85  GetGroundHeight( &pathEndPosition, &pathEndPosition.z );
86 
87  //
88  // Compute shortest path to goal
89  //
90  CNavArea *closestArea;
91  bool pathToGoalExists = NavAreaBuildPath( startArea, goalArea, goal, costFunc, &closestArea );
92 
93  CNavArea *effectiveGoalArea = (pathToGoalExists) ? goalArea : closestArea;
94 
95  //
96  // Build path by following parent links
97  //
98 
99  // get count
100  int count = 0;
101  CNavArea *area;
102  for( area = effectiveGoalArea; area; area = area->GetParent() )
103  ++count;
104 
105  // save room for endpoint
106  if (count > MAX_PATH_SEGMENTS-1)
107  count = MAX_PATH_SEGMENTS-1;
108 
109  if (count == 0)
110  return false;
111 
112  if (count == 1)
113  {
114  BuildTrivialPath( start, goal );
115  return true;
116  }
117 
118  // build path
119  m_segmentCount = count;
120  for( area = effectiveGoalArea; count && area; area = area->GetParent() )
121  {
122  --count;
123  m_path[ count ].area = area;
124  m_path[ count ].how = area->GetParentHow();
125  }
126 
127  // compute path positions
128  if (ComputePathPositions() == false)
129  {
130  //PrintIfWatched( "Error building path\n" );
131  Invalidate();
132  return false;
133  }
134 
135  // append path end position
136  m_path[ m_segmentCount ].area = effectiveGoalArea;
137  m_path[ m_segmentCount ].pos = pathEndPosition;
138  m_path[ m_segmentCount ].ladder = NULL;
139  m_path[ m_segmentCount ].how = NUM_TRAVERSE_TYPES;
140  ++m_segmentCount;
141 
142  return true;
143  }
