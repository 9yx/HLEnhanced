51 bool CallFunction( CALLABLE& callable, CallFlags_t flags, const ARGS& args )
52 {
53  auto pContext = callable.GetContext().GetContext();
54 
55  assert( pContext );
56 
57  if( !pContext )
58  return false;
59 
60  auto& function = callable.GetFunction();
61 
62  auto result = pContext->Prepare( &function );
63 
64  if( result < 0 )
65  {
66  return false;
67  }
68 
69  if( !callable.PreSetArguments() )
70  return false;
71 
72  auto success = ctx::SetArguments( function, *pContext, args );
73 
74  if( !success )
75  return false;
76 
77  if( !callable.PreExecute() )
78  return false;
79 
80  result = pContext->Execute();
81 
82  if( !callable.PostExecute( result ) )
83  return false;
84 
85  //TODO: check for errors. - Solokiller
86 
87  return result >= 0;
