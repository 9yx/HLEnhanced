372 inline unsigned short CSaveRestoreSegment::FindCreateSymbol( const char *pszToken )
373 {
374  unsigned short hash = (unsigned short)(HashString( pszToken ) % (unsigned)tokenCount );
375 
376 #if _DEBUG
377  static int tokensparsed = 0;
378  tokensparsed++;
379  if ( !tokenCount || !pTokens )
380  {
381  AssertMsg( 0, ("No token table array in TokenHash()!") );
382  }
383 #endif
384 
385  for ( int i=0; i<tokenCount; i++ )
386  {
387 #if _DEBUG
388  static bool beentheredonethat = false;
389  if ( i > 50 && !beentheredonethat )
390  {
391  beentheredonethat = true;
392  AssertMsg( 0, ("CSaveRestoreBuffer::TokenHash() is getting too full!" ) );
393  }
394 #endif
395 
396  int index = hash + i;
397  if ( index >= tokenCount )
398  index -= tokenCount;
399 
400  if ( !pTokens[index] || strcmp( pszToken, pTokens[index] ) == 0 )
401  {
402  pTokens[index] = (char *)pszToken;
403  return index;
404  }
405  }

422 #ifndef _WIN32
423 inline unsigned CSaveRestoreSegment::_rotr ( unsigned val, int shift)
424 {
425  register unsigned lobit; /* non-zero means lo bit set */
426  register unsigned num = val; /* number to rotate */
427 
428  shift &= 0x1f; /* modulo 32 -- this will also make
429  negative shifts work */
430 
431  while (shift--)
432  {
433  lobit = num & 1; /* get high bit */
434  num >>= 1; /* shift right one bit */
435  if (lobit)
436  num |= 0x80000000; /* set hi bit if lo bit was set */
437  }
438 
439  return num;
440 }
441 #endif
