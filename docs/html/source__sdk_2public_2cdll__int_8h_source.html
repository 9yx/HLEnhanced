81 // Purpose: The engine reports to the client DLL what stage it's entering so the DLL can latch events
82 // and make sure that certain operations only happen during the right stages.
83 // The value for each stage goes up as you move through the frame so you can check ranges of values
84 // and if new stages get added in-between, the range is still valid.
85 //-----------------------------------------------------------------------------
86 enum ClientFrameStage_t
87 {
88  FRAME_UNDEFINED=-1, // (haven't run any frames yet)
89  FRAME_START,
90 
91  // A network packet is being recieved
92  FRAME_NET_UPDATE_START,
93  // Data has been received and we're going to start calling PostDataUpdate
94  FRAME_NET_UPDATE_POSTDATAUPDATE_START,
95  // Data has been received and we've called PostDataUpdate on all data recipients
96  FRAME_NET_UPDATE_POSTDATAUPDATE_END,
97  // We've received all packets, we can now do interpolation, prediction, etc..
98  FRAME_NET_UPDATE_END,
99 
100  // We're about to start rendering the scene
101  FRAME_RENDER_START,
102  // We've finished rendering the scene.
103  FRAME_RENDER_END
104 };
