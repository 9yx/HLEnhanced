341 I CUtlLinkedList<T,I>::AllocInternal( bool multilist )
342 {
343  I elem;
344  if (m_FirstFree == InvalidIndex())
345  {
346  // Nothing in the free list; add.
347  // Since nothing is in the free list, m_TotalElements == total # of elements
348  // the list knows about.
349  if ((int)m_TotalElements == m_Memory.NumAllocated())
350  {
351  MEM_ALLOC_CREDIT_CLASS();

353  }
354 
355  Assert( m_TotalElements != InvalidIndex() );
356

358  ++m_TotalElements;
359

361  {
362  Error("CUtlLinkedList overflow!\n");
363  }
364  }
365  else
366  {
367  elem = m_FirstFree;
368  m_FirstFree = InternalElement(m_FirstFree).m_Next;
369  }
370

372  InternalElement(elem).m_Next = InternalElement(elem).m_Previous = elem;
373  else

377

534 template <class T, class I>
535 void CUtlLinkedList<T,I>::RemoveAll()
536 {
537  if (m_TotalElements == 0)
538  return;
539 
540  // Put everything into the free list
541  I prev = InvalidIndex();
542  for (int i = (int)m_TotalElements; --i >= 0; )
543  {
544  // Invoke the destructor
545  if (IsValidIndex((I)i))

554  }

557  m_FirstFree = 0;
558 
559  // Clear everything else out
560  m_Head = InvalidIndex();

574 template <class T, class I>
575 void CUtlLinkedList<T,I>::LinkBefore( I before, I elem )
576 {
577  Assert( IsValidIndex(elem) );
578 
579  // Unlink it if it's in the list at the moment
580  Unlink(elem);
581 
582  ListElem_t& newElem = InternalElement(elem);
583 
584  // The element *after* our newly linked one is the one we linked before.
585  newElem.m_Next = before;
586 
587  if (before == InvalidIndex())
588  {
589  // In this case, we're linking to the end of the list, so reset the tail
590  newElem.m_Previous = m_Tail;
591  m_Tail = elem;
592  }
593  else
594  {
595  // Here, we're not linking to the end. Set the prev pointer to point to
596  // the element we're linking.
597  Assert( IsInList(before) );
598  ListElem_t& beforeElem = InternalElement(before);
599  newElem.m_Previous = beforeElem.m_Previous;
600  beforeElem.m_Previous = elem;
601  }
602 
603  // Reset the head if we linked to the head of the list
604  if (newElem.m_Previous == InvalidIndex())
605  m_Head = elem;
606  else
607  InternalElement(newElem.m_Previous).m_Next = elem;
608 
609  // one more element baby
610  ++m_ElementCount;
611 }

652 template <class T, class I>
653 void CUtlLinkedList<T,I>::Unlink( I elem )
654 {
655  Assert( IsValidIndex(elem) );
656  if (IsInList(elem))
657  {
658  ListElem_t *pBase = m_Memory.Base();
659  ListElem_t *pOldElem = &pBase[elem];
660 
661  // If we're the first guy, reset the head
662  // otherwise, make our previous node's next pointer = our next
663  if ( pOldElem->m_Previous != INVALID_LLIST_IDX )
664  {
665  pBase[pOldElem->m_Previous].m_Next = pOldElem->m_Next;
666  }
667  else
668  {
669  m_Head = pOldElem->m_Next;
670  }
671 
672  // If we're the last guy, reset the tail
673  // otherwise, make our next node's prev pointer = our prev
674  if ( pOldElem->m_Next != INVALID_LLIST_IDX )
675  {
676  pBase[pOldElem->m_Next].m_Previous = pOldElem->m_Previous;
677  }
678  else
679  {
680  m_Tail = pOldElem->m_Previous;
681  }
682 
683  // This marks this node as not in the list,
684  // but not in the free list either
685  pOldElem->m_Previous = pOldElem->m_Next = elem;
686 
687  // One less puppy
688  --m_ElementCount;
689  }
690 }
