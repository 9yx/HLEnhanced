63  // If this is true, then it won't look at -vproject, -game, or the vproject environment variable

65  //
66  // (default: false)
67  bool m_bOnlyUseDirectoryName;
68 
69  // If this is true, then:
70  // 1. It will set the environment variables that steam.dll looks at for startup info.
71  // 2. It will look for ToolsAppId in the gameinfo.txt file and load the
72  // steam caches associated with that cache if it's there. This is so apps like Hammer and hlmv
73  // can load the main steam caches (like for Counter-Strike or Half-Life 2), and also load the
74  // caches that include tools-specific materials (materials\editor, materials\debug, etc).
75  //
76  // (default: true - should be FALSE for the engine)
77  bool m_bToolsMode;
78 
79  // If this is true, and m_bToolsMode is false, then it will append the path to steam.dll to the
80  // PATH environment variable. This makes it so you can run the engine under Steam without
81  // having to copy steam.dll up into your hl2.exe folder.
82  // (default: false)
83  bool m_bSetSteamDLLPath;
84 
85  // Are we loading the Steam filesystem? This should be the same value that
86  // FileSystem_GetFileSystemDLLName gave you.
87  bool m_bSteam;
88 
89  // If this is true, then it won't look for a gameinfo.txt.
90  //
91  // (default: false)
92  bool m_bNoGameInfo;

94 // Outputs (if it returns FS_OK).
