411 //-----------------------------------------------------------------------------
412 //
413 // An alternative mutex that is useful for cases when thread contention is
414 // rare, but a mutex is required. Instances should be declared volatile.
415 // Sleep of 0 may not be sufficient to keep high priority threads from starving
416 // lesser threads. This class is not a suitable replacement for a critical
417 // section if the resource contention is high.
418 //
419 //-----------------------------------------------------------------------------
420 
421 class TT_CLASS CThreadFastMutex
422 {
423 public:
424  CThreadFastMutex()
425  : m_ownerID( 0 ),
426  m_depth( 0 )
427  {
428  }
429 
430 private:
431  bool TryLock( const uint32 threadId ) volatile
432  {
433  if ( threadId != m_ownerID && ThreadInterlockedCompareExchange( (volatile long *)&m_ownerID, (long)threadId, 0 ) != 0 )
434  return false;
435 
436  m_depth++;
437  return true;
438  }
439 
440 public:
441  bool TryLock() volatile
442  {
443  return TryLock( ThreadGetCurrentId() );
444  }
445 
446  void Lock( unsigned nSpinSleepTime = 1 ) volatile
447  {
448  const uint32 threadId = ThreadGetCurrentId();
449 
450 
451  if ( TryLock( threadId ) )
452  {
453  return;
454  }
455 
456  ThreadPause();
457 
458  for ( int i = 16; i > 0; --i )
459  {
460  if ( TryLock( threadId ) )
461  {
462  return;
463  }
464 
465  ThreadPause();
466  ThreadSleep( 0 );
467  }
468 
469  while ( !TryLock( threadId ) )
470  {
471  ThreadPause();
472  ThreadSleep( nSpinSleepTime );
473  };
474 
475 #ifdef _DEBUG
476  if ( m_ownerID != ThreadGetCurrentId() )
477  DebuggerBreak();
478 #endif
479  }
480 
481  void Unlock() volatile
482  {
483 #ifdef _DEBUG
484  if ( m_ownerID != ThreadGetCurrentId() )
485  DebuggerBreak();
486 #endif
487  m_depth--;
488  if ( !m_depth )
489  m_ownerID = 0;
490  }
491 
492  bool TryLock() const volatile { return (const_cast<CThreadFastMutex *>(this))->TryLock(); }
493  void Lock(unsigned nSpinSleepTime = 1 ) const volatile { (const_cast<CThreadFastMutex *>(this))->Lock( nSpinSleepTime ); }
494  void Unlock() const volatile { (const_cast<CThreadFastMutex *>(this))->Unlock(); }
495 
496  // To match regular CThreadMutex:
497  bool AssertOwnedByCurrentThread() { return true; }
498  void SetTrace( bool ) {}
499 private:
500  volatile uint32 m_ownerID;
501  volatile int m_depth;
502 };

984 inline void CThreadMutex::Lock()
