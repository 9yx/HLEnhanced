35 template <class REFCOUNTED_ITEM_PTR>
36 inline int SafeRelease( REFCOUNTED_ITEM_PTR &pRef )
37 {
38  // Use funny syntax so that this works on "auto pointers"
39  REFCOUNTED_ITEM_PTR *ppRef = &pRef;
40  if ( *ppRef )
41  {
42  int result = (*ppRef)->Release();
43  *ppRef = NULL;
44  return result;
45  }
46  return 0;
47 }
