232 /* SDL stdinc inline functions:
233 
234  The theory here is that by default we forcibly inline what we can, and your
235  app will use the inline version by default. However we expose a non-inline
236  version too, so the symbol is always available in the library even if your app
237  bypassed the inline version. The SDL_*_inline versions aren't guaranteed to
238  exist, so never call them directly; use SDL_* instead, and trust the system
239  to give you the right thing.
240 
241  The benefit here is that you can dlsym() these functions, which you
242  couldn't if you had macros, you can link against a foreign build of SDL
243  even if you configured differently, and you can drop the unconfigured SDL
244  headers into a project without #defining HAVE_MALLOC (etc) and still link.
245 */

332 SDL_FORCE_INLINE void SDL_memset4(void *dst, int val, size_t len)
333 {
334 #if defined(__GNUC__) && defined(i386)
335  int u0, u1, u2;
336  __asm__ __volatile__ (
337  "cld \n\t"
338  "rep ; stosl \n\t"
339  : "=&D" (u0), "=&a" (u1), "=&c" (u2)
340  : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, len))
341  : "memory"
342  );
343 /* !!! FIXME: amd64? */
344 #else
345  size_t _n = (len + 3) / 4;
346  Uint32 *_p = SDL_static_cast(Uint32 *, dst);
347  Uint32 _val = (val);
348  if (len == 0)
349  return;
350  switch (len % 4)
351  {
352  case 0: do { *_p++ = _val;
353  case 3: *_p++ = _val;
354  case 2: *_p++ = _val;
355  case 1: *_p++ = _val;
356  } while ( --_n );
357  }
358 #endif
359 }
