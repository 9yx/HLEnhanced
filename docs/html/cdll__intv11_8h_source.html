91 //-----------------------------------------------------------------------------
92 // Purpose: The engine reports to the client DLL what stage it's entering so the DLL can latch events
93 // and make sure that certain operations only happen during the right stages.
94 // The value for each stage goes up as you move through the frame so you can check ranges of values
95 // and if new stages get added in-between, the range is still valid.
96 //-----------------------------------------------------------------------------
97 enum ClientFrameStage_t
98 {
99  FRAME_UNDEFINED=-1, // (haven't run any frames yet)
100  FRAME_START,
101 
102  // A network packet is being recieved
103  FRAME_NET_UPDATE_START,
104  // Data has been received and we're going to start calling PostDataUpdate
105  FRAME_NET_UPDATE_POSTDATAUPDATE_START,
106  // Data has been received and we've called PostDataUpdate on all data recipients
107  FRAME_NET_UPDATE_POSTDATAUPDATE_END,
108  // We've received all packets, we can now do interpolation, prediction, etc..
109  FRAME_NET_UPDATE_END,
110 
111  // We're about to start rendering the scene
112  FRAME_RENDER_START,
113  // We've finished rendering the scene.
114  FRAME_RENDER_END
115 };
