374 template <class T, class I>
375 I CUtlMultiList<T,I>::Alloc( )
376 {
377  I elem;
378  if (m_FirstFree == InvalidIndex())
379  {
380  // Nothing in the free list; add.
381  // Since nothing is in the free list, m_TotalElements == total # of elements
382  // the list knows about.
383  if (m_MaxElementIndex == m_Memory.NumAllocated())
384  {
385  m_Memory.Grow();
386  ResetDbgInfo();
387  }
388 
389  elem = (I)m_MaxElementIndex;
390  ++m_MaxElementIndex;
391  }
392  else
393  {
394  elem = m_FirstFree;
395  m_FirstFree = InternalElement(m_FirstFree).m_Next;
396  }

399  InternalElement(elem).m_Next = InternalElement(elem).m_Previous = elem;
400

435 template <class T, class I>
436 void CUtlMultiList<T,I>::LinkBefore( ListHandle_t list, I before, I elem )
437 {
438  Assert( IsValidIndex(elem) && IsValidList(list) );
439 
440  // Unlink it if it's in the list at the moment
441  Unlink(list, elem);
442 
443  ListElem_t& newElem = InternalElement(elem);
444 
445  // The element *after* our newly linked one is the one we linked before.
446  newElem.m_Next = before;
447 
448  if (before == InvalidIndex())
449  {
450  // In this case, we're linking to the end of the list, so reset the tail
451  newElem.m_Previous = m_List[list].m_Tail;
452  m_List[list].m_Tail = elem;
453  }
454  else
455  {
456  // Here, we're not linking to the end. Set the prev pointer to point to
457  // the element we're linking.
458  Assert( IsInList(before) );
459  ListElem_t& beforeElem = InternalElement(before);
460  newElem.m_Previous = beforeElem.m_Previous;
461  beforeElem.m_Previous = elem;
462  }
463 
464  // Reset the head if we linked to the head of the list
465  if (newElem.m_Previous == InvalidIndex())
466  m_List[list].m_Head = elem;
467  else
468  InternalElement(newElem.m_Previous).m_Next = elem;
469 
470  // one more element baby
471  ++m_List[list].m_Count;
472 
473  // Store the element into the list
474  if (m_pElementList)
475  m_pElementList[elem] = list;
476 }

520 template <class T, class I>
521 void CUtlMultiList<T,I>::Unlink( ListHandle_t list, I elem )
522 {
523  Assert( IsValidIndex(elem) && IsValidList(list) );
524 
525  if (IsInList(elem))
526  {
527  // Make sure the element is in the right list
528  Assert( IsElementInList( list, elem ) );
529  ListElem_t& oldElem = InternalElement(elem);
530 
531  // If we're the first guy, reset the head
532  // otherwise, make our previous node's next pointer = our next
533  if (oldElem.m_Previous != InvalidIndex())
534  InternalElement(oldElem.m_Previous).m_Next = oldElem.m_Next;
535  else
536  m_List[list].m_Head = oldElem.m_Next;
537 
538  // If we're the last guy, reset the tail
539  // otherwise, make our next node's prev pointer = our prev
540  if (oldElem.m_Next != InvalidIndex())
541  InternalElement(oldElem.m_Next).m_Previous = oldElem.m_Previous;
542  else
543  m_List[list].m_Tail = oldElem.m_Previous;
544 
545  // This marks this node as not in the list,
546  // but not in the free list either
547  oldElem.m_Previous = oldElem.m_Next = elem;
548 
549  // One less puppy
550  --m_List[list].m_Count;
551 
552  // Store the element into the list
553  if (m_pElementList)
554  m_pElementList[elem] = m_List.InvalidIndex();
555  }
556 }

690 template <class T, class I>
691 void CUtlMultiList<T,I>::RemoveAll()
692 {
693  if (m_MaxElementIndex == 0)
694  return;
695 
696  // Put everything into the free list
697  I prev = InvalidIndex();
698  for (int i = (int)m_MaxElementIndex; --i >= 0; )
699  {
700  // Invoke the destructor
701  if (IsValidIndex((I)i))
702  Destruct( &Element((I)i) );
703 
704  // next points to the next free list item
705  InternalElement((I)i).m_Next = prev;
706 
707  // Indicates it's in the free list
708  InternalElement((I)i).m_Previous = (I)i;
709  prev = (I)i;
710  }
711 
712  // First free points to the first element
713  m_FirstFree = 0;
714 
715  // Clear everything else out
716  for (I list = m_List.Head(); list != m_List.InvalidIndex(); list = m_List.Next(list) )
717  {
718  m_List[list].m_Head = InvalidIndex();
719  m_List[list].m_Tail = InvalidIndex();
720  m_List[list].m_Count = 0;
721  }
722 
723  m_TotalElements = 0;
724 }
