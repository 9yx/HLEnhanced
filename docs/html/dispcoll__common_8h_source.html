440 {
441  int nIndex = 0;
442 
443  // Interleave bits from the x and y values to create the index
444  int iShift;
445  for( iShift = 0; x != 0; iShift += 2, x >>= 1 )
446  {
447  nIndex |= ( x & 1 ) << iShift;

449
