20 // class CUtlSortVector:
21 // description:
22 // This in an sorted order-preserving vector. Items may be inserted or removed
23 // at any point in the vector. When an item is inserted, all elements are
24 // moved down by one element using memmove. When an item is removed, all
25 // elements are shifted back down. Items are searched for in the vector
26 // using a binary search technique. Clients must pass in a Less() function
27 // into the constructor of the vector to determine the sort order.
28 //-----------------------------------------------------------------------------

120 int CUtlSortVector<T, LessFunc>::Insert( const T& src )
121 {
122  AssertFatal( !m_bNeedsSort );
123 
124  int pos = FindLessOrEqual( src ) + 1;
125  GrowVector();
126  ShiftElementsRight(pos);
127  CopyConstruct<T>( &Element(pos), src );
128  return pos;
129 }

162 template <class T, class LessFunc>
163 int CUtlSortVector<T, LessFunc>::SplitList( LessFunc& less, int nLower, int nUpper )
164 {
165  int nLeft = nLower + 1;
166  int nRight = nUpper;
167 
168  const T& val = Element( nLower );
169 
170  while ( nLeft <= nRight )
171  {
172  while ( nLeft <= nRight && !less.Less( val, Element( nLeft ), m_pLessContext ) )
173  ++nLeft;
174  while ( nLeft <= nRight && !less.Less( Element( nRight ), val, m_pLessContext ) )
175  --nRight;
176 
177  if ( nLeft < nRight )
178  {
179  Swap( nLeft, nRight );
180  ++nLeft;
181  --nRight;
182  }
183  }
184 
185  Swap( nLower, nRight );
186  return nRight;
187 }

204 int CUtlSortVector<T, LessFunc>::Find( const T& src ) const
205 {
206  AssertFatal( !m_bNeedsSort );
207 
208  LessFunc less;
209 
210  int start = 0, end = Count() - 1;
211  while (start <= end)
212  {
213  int mid = (start + end) >> 1;
214  if ( less.Less( Element(mid), src, m_pLessContext ) )
215  {
216  start = mid + 1;
217  }
218  else if ( less.Less( src, Element(mid), m_pLessContext ) )
219  {
220  end = mid - 1;
221  }
222  else
223  {
224  return mid;
225  }
226  }
