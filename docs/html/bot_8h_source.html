36 template <class T> T * CreateBot( const BotProfile *profile )
37 {
38  edict_t * pentBot;
39 
40  if ( UTIL_ClientsInGame() >= gpGlobals->maxClients )
41  {
42  CONSOLE_ECHO( "Unable to create bot: Server is full (%d/%d clients).\n", UTIL_ClientsInGame(), gpGlobals->maxClients );
43  return NULL;
44  }
45 
46  char netname[64];
47 
48  UTIL_ConstructBotNetName(netname, 64, profile);
49 
50  pentBot = CREATE_FAKE_CLIENT( netname );
51 
52  if ( FNullEnt( pentBot ) )
53  {
54  CONSOLE_ECHO( "Unable to create bot: pfnCreateFakeClient() returned null.\n" );
55  return NULL;
56  }
57  else
58  {
59  T * pBot = NULL;
60 
61  FREE_PRIVATE( pentBot );
62  pBot = GetClassPtr( (T *)VARS( pentBot ) );
63 
64  // initialize the bot
65  pBot->Initialize( profile );
66 
67  return pBot;
68  }
69 }

341 {
342  Vector toOther = other->pev->origin - pev->origin;

355 inline bool CBot::IsPlayerLookingAtMe( CBasePlayer *other ) const
356 {
357  Vector toOther = other->pev->origin - pev->origin;
358  toOther.NormalizeInPlace();
359 
360  // compute the unit vector along our other player's
361  UTIL_MakeVectors( other->pev->v_angle + other->pev->punchangle );
362  Vector otherDir = gpGlobals->v_forward;
363 
364  // other player must be pointing nearly right at us to be "looking at" us
365  const float lookAtCos = 0.9f;
366  if (otherDir.x * toOther.x + otherDir.y * toOther.y < -lookAtCos)
367  {
368  // other player must have clear line of sight to us to be "looking at" us
369  Vector vec(other->EyePosition());
370  if (IsVisible( &vec ))
371  return true;
372  }
373 
374  return false;
375 }
