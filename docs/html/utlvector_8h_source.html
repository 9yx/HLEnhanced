29 // The CUtlVector class:
30 // A growable array class which doubles in size by default.
31 // It will always keep all elements consecutive in memory, and may move the
32 // elements around in memory (via a PvRealloc) when elements are inserted or
33 // removed. Clients should therefore refer to the elements of the vector
34 // by index (they should *never* maintain pointers to elements in the vector).
35 //-----------------------------------------------------------------------------
