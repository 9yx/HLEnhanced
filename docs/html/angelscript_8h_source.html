600  bool hasConstructor = std::is_default_constructible<T>::value && !std::is_trivially_default_constructible<T>::value;
601  bool hasDestructor = std::is_destructible<T>::value && !std::is_trivially_destructible<T>::value;
602  bool hasAssignmentOperator = std::is_copy_assignable<T>::value && !std::is_trivially_copy_assignable<T>::value;
603  bool hasCopyConstructor = std::is_copy_constructible<T>::value && !std::is_trivially_copy_constructible<T>::value;
604 #elif defined(__GNUC__) && (__GNUC__ > 4 || (__GNUC__ == 4 && __GNUC_MINOR__ >= 8))
605  // gnuc 4.8 is using a mix of C++11 standard and pre-standard templates
606  bool hasConstructor = std::is_default_constructible<T>::value && !std::has_trivial_default_constructor<T>::value;
607  bool hasDestructor = std::is_destructible<T>::value && !std::is_trivially_destructible<T>::value;
608  bool hasAssignmentOperator = std::is_copy_assignable<T>::value && !std::has_trivial_copy_assign<T>::value;
609  bool hasCopyConstructor = std::is_copy_constructible<T>::value && !std::has_trivial_copy_constructor<T>::value;
610 #else
611  // All other compilers and versions are assumed to use non C++11 compliant code until proven otherwise
612  // Not fully C++11 compliant. The has_trivial checks were used while the standard was still
613  // being elaborated, but were then removed in favor of the above is_trivially checks
614  // http://stackoverflow.com/questions/12702103/writing-code-that-works-when-has-trivial-destructor-is-defined-instead-of-is
615  // https://github.com/mozart/mozart2/issues/51
616  bool hasConstructor = std::is_default_constructible<T>::value && !std::has_trivial_default_constructor<T>::value;
617  bool hasDestructor = std::is_destructible<T>::value && !std::has_trivial_destructor<T>::value;
618  bool hasAssignmentOperator = std::is_copy_assignable<T>::value && !std::has_trivial_copy_assign<T>::value;
619  bool hasCopyConstructor = std::is_copy_constructible<T>::value && !std::has_trivial_copy_constructor<T>::value;
620 #endif
621  bool isFloat = std::is_floating_point<T>::value;
622  bool isPrimitive = std::is_integral<T>::value || std::is_pointer<T>::value || std::is_enum<T>::value;
623  bool isClass = std::is_class<T>::value;
624  bool isArray = std::is_array<T>::value;
625 
626  if( isFloat )
627  return asOBJ_APP_FLOAT;
628  if( isPrimitive )
629  return asOBJ_APP_PRIMITIVE;
630 
631  if( isClass )
632  {
633  asDWORD flags = asOBJ_APP_CLASS;
634  if( hasConstructor )
635  flags |= asOBJ_APP_CLASS_CONSTRUCTOR;
636  if( hasDestructor )
637  flags |= asOBJ_APP_CLASS_DESTRUCTOR;
638  if( hasAssignmentOperator )
639  flags |= asOBJ_APP_CLASS_ASSIGNMENT;
640  if( hasCopyConstructor )
641  flags |= asOBJ_APP_CLASS_COPY_CONSTRUCTOR;
642  return flags;
643  }
644 
645  if( isArray )
646  return asOBJ_APP_ARRAY;
647 
648  // Unknown type traits
649  return 0;
650 }
