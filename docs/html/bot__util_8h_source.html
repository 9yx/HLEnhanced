247 inline bool IsIntersecting2D( const Vector &startA, const Vector &endA,
248  const Vector &startB, const Vector &endB,
249  Vector *result = NULL )
250 {
251  float denom = (endA.x - startA.x) * (endB.y - startB.y) - (endA.y - startA.y) * (endB.x - startB.x);
252  if (denom == 0.0f)
253  {
254  // parallel
255  return false;
256  }
257 
258  float numS = (startA.y - startB.y) * (endB.x - startB.x) - (startA.x - startB.x) * (endB.y - startB.y);
259  if (numS == 0.0f)
260  {
261  // coincident
262  return true;
263  }
264 
265  float numT = (startA.y - startB.y) * (endA.x - startA.x) - (startA.x - startB.x) * (endA.y - startA.y);
266 
267  float s = numS / denom;
268  if (s < 0.0f || s > 1.0f)
269  {
270  // intersection is not within line segment of startA to endA
271  return false;
272  }
273 
274  float t = numT / denom;
275  if (t < 0.0f || t > 1.0f)
276  {
277  // intersection is not within line segment of startB to endB
278  return false;
279  }
280 
281  // compute intesection point
282  if (result)
283  *result = startA + s * (endA - startA);
284 
285  return true;
286 }
