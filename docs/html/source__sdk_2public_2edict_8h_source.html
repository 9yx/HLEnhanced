268 inline void CBaseEdict::StateChanged( unsigned short offset )
269 {
270  if ( m_fStateFlags & FL_FULL_EDICT_CHANGED )
271  return;
272 
273  m_fStateFlags |= FL_EDICT_CHANGED;
274 
275  IChangeInfoAccessor *accessor = GetChangeAccessor();
276 
277  if ( accessor->GetChangeInfoSerialNumber() == g_pSharedChangeInfo->m_iSerialNumber )
278  {
279  // Ok, I still own this one.
280  CEdictChangeInfo *p = &g_pSharedChangeInfo->m_ChangeInfos[accessor->GetChangeInfo()];
281 
282  // Now add this offset to our list of changed variables.
283  for ( unsigned short i=0; i < p->m_nChangeOffsets; i++ )
284  if ( p->m_ChangeOffsets[i] == offset )
285  return;
286 
287  if ( p->m_nChangeOffsets == MAX_CHANGE_OFFSETS )
288  {
289  // Invalidate our change info.
290  accessor->SetChangeInfoSerialNumber( 0 );
291  m_fStateFlags |= FL_FULL_EDICT_CHANGED; // So we don't get in here again.
292  }
293  else
294  {
295  p->m_ChangeOffsets[p->m_nChangeOffsets++] = offset;
296  }
297  }
298  else
299  {
300  if ( g_pSharedChangeInfo->m_nChangeInfos == MAX_EDICT_CHANGE_INFOS )
301  {
302  // Shucks.. have to mark the edict as fully changed because we don't have room to remember this change.
303  accessor->SetChangeInfoSerialNumber( 0 );
304  m_fStateFlags |= FL_FULL_EDICT_CHANGED;
305  }
306  else
307  {
308  // Get a new CEdictChangeInfo and fill it out.
309  accessor->SetChangeInfo( g_pSharedChangeInfo->m_nChangeInfos );
310  g_pSharedChangeInfo->m_nChangeInfos++;
311 
312  accessor->SetChangeInfoSerialNumber( g_pSharedChangeInfo->m_iSerialNumber );
313 
314  CEdictChangeInfo *p = &g_pSharedChangeInfo->m_ChangeInfos[accessor->GetChangeInfo()];
315  p->m_ChangeOffsets[0] = offset;
316  p->m_nChangeOffsets = 1;
317  }
318  }
319 }
