22 // ------------------------------------------------------------------------ //
23 // Send proxies can be used to convert a variable into a networkable type
24 // (a good example is converting an edict pointer into an integer index).
25 
26 // These allow you to translate data. For example, if you had a user-entered
27 // string number like "10" (call the variable pUserStr) and wanted to encode
28 // it as an integer, you would use a SendPropInt32 and write a proxy that said:
29 // pOut->m_Int = atoi(pUserStr);
30 
31 // pProp : the SendProp that has the proxy
32 // pStructBase : the base structure (like CBaseEntity*).
33 // pData : the address of the variable to proxy.
34 // pOut : where to output the proxied value.
35 // iElement : the element index if this data is part of an array (or 0 if not).
36 // objectID : entity index for debugging purposes.
37 
38 // Return false if you don't want the engine to register and send a delta to
39 // the clients for this property (regardless of whether it actually changed or not).
40 // ------------------------------------------------------------------------ //
41 typedef void (*SendVarProxyFn)( const SendProp *pProp, const void *pStructBase, const void *pData, DVariant *pOut, int iElement, int objectID );

521 // ------------------------------------------------------------------------------------------------------ //
522 // Use BEGIN_SEND_TABLE if you want to declare a SendTable and have it inherit all the properties from
523 // its base class. There are two requirements for this to work:
524 
525 // 1. Its base class must have a static SendTable pointer member variable called m_pClassSendTable which
526 // points to its send table. The DECLARE_SERVERCLASS and IMPLEMENT_SERVERCLASS macros do this automatically.
527 
528 // 2. Your class must typedef its base class as BaseClass. So it would look like this:
529 // class Derived : public CBaseEntity
530 // {
531 // typedef CBaseEntity BaseClass;
532 // };
533 
534 // If you don't want to interit a base class's properties, use BEGIN_SEND_TABLE_NOBASE.
535 // ------------------------------------------------------------------------------------------------------ //
536 #define BEGIN_SEND_TABLE(className, tableName) \
537  BEGIN_SEND_TABLE_NOBASE(className, tableName) \
538  SendPropDataTable("baseclass", 0, className::BaseClass::m_pClassSendTable, SendProxy_DataTableToDataTable),
539 
540 #define BEGIN_SEND_TABLE_NOBASE(className, tableName) \
541  template <typename T> int ServerClassInit(T *); \
542  namespace tableName { \
543  struct ignored; \
544  } \
545  template <> int ServerClassInit<tableName::ignored>(tableName::ignored *); \
546  namespace tableName { \
547  SendTable g_SendTable;\
548  int g_SendTableInit = ServerClassInit((tableName::ignored *)NULL); \
549  } \
550  template <> int ServerClassInit<tableName::ignored>(tableName::ignored *) \
551  { \
552  typedef className currentSendDTClass; \
553  static char *g_pSendTableName = #tableName; \
554  SendTable &sendTable = tableName::g_SendTable; \
555  static SendProp g_SendProps[] = { \
556  SendPropInt("should_never_see_this", 0, sizeof(int)), // It adds a dummy property at the start so you can define "empty" SendTables.
557 
558 #define END_SEND_TABLE() \
559  };\
560  sendTable.Construct(g_SendProps+1, sizeof(g_SendProps) / sizeof(SendProp) - 1, g_pSendTableName);\
561  return 1; \
562  }
