455 template <typename T>
456 inline T WordSwapC( T w )
457 {
458  uint16 temp;
459 
460  temp = ((*((uint16 *)&w) & 0xff00) >> 8);
461  temp |= ((*((uint16 *)&w) & 0x00ff) << 8);
462 
463  return *((T*)&temp);
464 }
465 
466 template <typename T>
467 inline T DWordSwapC( T dw )
468 {
469  uint32 temp;
470 
471  temp = *((uint32 *)&dw) >> 24;
472  temp |= ((*((uint32 *)&dw) & 0x00FF0000) >> 8);
473  temp |= ((*((uint32 *)&dw) & 0x0000FF00) << 8);
474  temp |= ((*((uint32 *)&dw) & 0x000000FF) << 24);
475 
476  return *((T*)&temp);
477 }
