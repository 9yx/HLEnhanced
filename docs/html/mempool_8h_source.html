211 {

213  SetDefLessFunc( freeBlocks );
214 
215  void *pCurFree = m_pHeadOfFreeList;
216  while ( pCurFree != NULL )
217  {
218  freeBlocks.Insert( pCurFree );
219  pCurFree = *((void**)pCurFree);
220  }
221 
222  for( CBlob *pCur=m_BlobHead.m_pNext; pCur != &m_BlobHead; pCur=pCur->m_pNext )
223  {
224  T *p = (T *)pCur->m_Data;
225  T *pLimit = (T *)(pCur->m_Data + pCur->m_NumBytes);
226  while ( p < pLimit )
227  {
228  if ( freeBlocks.Find( p ) == freeBlocks.InvalidIndex() )
229  {
230  Destruct( p );
231  }
232  p++;
233  }
234  }

288 inline void *CAlignedMemPool<ITEM_SIZE, ALIGNMENT, CHUNK_SIZE, CAllocator, COMPACT_THRESHOLD>::Alloc()
289 {
290  if ( !m_pFirstFree )
291  {
292  FreeBlock_t *pNew = (FreeBlock_t *)m_Allocator.Alloc( CHUNK_SIZE );
293  Assert( (unsigned)pNew % ALIGNMENT == 0 );
294  m_Chunks.AddToTail( pNew );
295  m_nFree = CHUNK_SIZE / BLOCK_SIZE;
296  m_pFirstFree = pNew;
297  for ( int i = 0; i < m_nFree - 1; i++ )
298  {
299  pNew->pNext = pNew + 1;
300  pNew++;
301  }
302  pNew->pNext = NULL;
303  }
304

306  m_pFirstFree = m_pFirstFree->pNext;
307  m_nFree--;

312 template <int ITEM_SIZE, int ALIGNMENT, int CHUNK_SIZE, class CAllocator, int COMPACT_THRESHOLD >
313 inline void CAlignedMemPool<ITEM_SIZE, ALIGNMENT, CHUNK_SIZE, CAllocator, COMPACT_THRESHOLD>::Free( void *p )
314 {
315  // Insertion sort to encourage allocation clusters in chunks
316  FreeBlock_t *pFree = ((FreeBlock_t *)p);
317  FreeBlock_t *pCur = m_pFirstFree;
318  FreeBlock_t *pPrev = NULL;
319 
320  while ( pCur && pFree > pCur )
321  {
322  pPrev = pCur;
323  pCur = pCur->pNext;
324  }
325 
326  pFree->pNext = pCur;
327 
328  if ( pPrev )
329  {
330  pPrev->pNext = pFree;
331  }
332  else
333  {
334  m_pFirstFree = pFree;
335  }
336  m_nFree++;
337 
338  if ( m_nFree >= ( CHUNK_SIZE / BLOCK_SIZE ) * COMPACT_THRESHOLD )
339  {
340  float time = Plat_FloatTime();
341  float compactTime = ( m_nFree >= ( CHUNK_SIZE / BLOCK_SIZE ) * COMPACT_THRESHOLD * 4 ) ? 15.0 : 30.0;
342  if ( m_TimeLastCompact > time || m_TimeLastCompact + compactTime < Plat_FloatTime() )
343  {
344  Compact();
345  m_TimeLastCompact = time;
346  }
347  }
348 }

357 inline void CAlignedMemPool<ITEM_SIZE, ALIGNMENT, CHUNK_SIZE, CAllocator, COMPACT_THRESHOLD>::Compact()
358 {
359  FreeBlock_t *pCur = m_pFirstFree;
360  FreeBlock_t *pPrev = NULL;
361 
362  m_Chunks.Sort( CompareChunk );
363 
364 #ifdef VALIDATE_ALIGNED_MEM_POOL
365  {
366  FreeBlock_t *p = m_pFirstFree;
367  while ( p )
368  {
369  if ( p->pNext && p > p->pNext )
370  {
371  __asm { int 3 }
372  }
373  p = p->pNext;
374  }
375 
376  for ( int i = 0; i < m_Chunks.Count(); i++ )
377  {
378  if ( i + 1 < m_Chunks.Count() )
379  {
380  if ( m_Chunks[i] > m_Chunks[i + 1] )
381  {
382  __asm { int 3 }
383  }
384  }
385  }
386  }
387 #endif
388 
389  int i;
390 
391  for ( i = 0; i < m_Chunks.Count(); i++ )
392  {
393  int nBlocksPerChunk = CHUNK_SIZE / BLOCK_SIZE;
394  FreeBlock_t *pChunkLimit = ((FreeBlock_t *)m_Chunks[i]) + nBlocksPerChunk;
395  int nFromChunk = 0;
396  if ( pCur == m_Chunks[i] )
397  {
398  FreeBlock_t *pFirst = pCur;
399  while ( pCur && pCur >= m_Chunks[i] && pCur < pChunkLimit )
400  {
401  pCur = pCur->pNext;
402  nFromChunk++;
403  }
404  pCur = pFirst;
405 
406  }
407 
408  while ( pCur && pCur >= m_Chunks[i] && pCur < pChunkLimit )
409  {
410  if ( nFromChunk != nBlocksPerChunk )
411  {
412  if ( pPrev )
413  {
414  pPrev->pNext = pCur;
415  }
416  else
417  {
418  m_pFirstFree = pCur;
419  }
420  pPrev = pCur;
421  }
422  else if ( pPrev )
423  {
424  pPrev->pNext = NULL;
425  }
426  else
427  {
428  m_pFirstFree = NULL;
429  }
430 
431  pCur = pCur->pNext;
432  }
433 
434  if ( nFromChunk == nBlocksPerChunk )
435  {
436  m_Allocator.Free( m_Chunks[i] );
437  m_nFree -= nBlocksPerChunk;
438  m_Chunks[i] = 0;
439  }
440  }
441 
442  for ( i = m_Chunks.Count() - 1; i >= 0 ; i-- )
443  {
444  if ( !m_Chunks[i] )
445  {
446  m_Chunks.FastRemove( i );
447  }
448  }
449 }
