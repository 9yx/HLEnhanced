300 //-----------------------------------------------------------------------------
301 // Node allocation/deallocation
302 //-----------------------------------------------------------------------------
303 template <class T, class I>
304 I CUtlNTree<T,I>::AllocInternal( )
305 {
306  I elem;
307  if ( m_FirstFree == INVALID_NTREE_IDX )
308  {
309  // Nothing in the free list; add.
310  // Since nothing is in the free list, m_MaxElementIndex == total # of elements
311  // the list knows about.
312  if ((int)m_MaxElementIndex == m_Memory.NumAllocated())
313  {
314  MEM_ALLOC_CREDIT_CLASS();
315  m_Memory.Grow();
316  }
317 
318  Assert( m_MaxElementIndex != INVALID_NTREE_IDX );
319 
320  elem = (I)m_MaxElementIndex;
321  ++m_MaxElementIndex;
322 
323  if ( elem == InvalidIndex() )
324  {
325  Error("CUtlNTree overflow!\n");
326  }
327  }
328  else
329  {
330  elem = m_FirstFree;
331  m_FirstFree = InternalNode( m_FirstFree ).m_NextSibling;
332  }
333 
334  Node_t &node = InternalNode( elem );
335  node.m_NextSibling = node.m_PrevSibling = node.m_Parent = node.m_FirstChild = INVALID_NTREE_IDX;
336  ResetDbgInfo();
337 
338  // one more element baby
339  ++m_ElementCount;
340 
341  return elem;
342 }

389 //-----------------------------------------------------------------------------
390 // Clears the tree
391 //-----------------------------------------------------------------------------
392 template <class T, class I>
393 void CUtlNTree<T,I>::RemoveAll()
394 {
395  if ( m_MaxElementIndex == 0 )
396  return;
397 
398  // Put everything into the free list (even unlinked things )
399  I prev = InvalidIndex();
400  for (int i = (int)m_MaxElementIndex; --i >= 0; prev = (I)i )
401  {
402  Node_t &node = InternalNode( i );
403  if ( IsInTree( i ) )
404  {
405  Destruct( &node.m_Element );
406  }
407 
408  node.m_NextSibling = prev;
409  node.m_PrevSibling = (I)i; // Marks it as being in the free list
410  node.m_Parent = node.m_FirstChild = INVALID_NTREE_IDX;
411  }
412 
413  // First free points to the first element
414  m_FirstFree = 0;
415 
416  // Clear everything else out
417  m_Root = INVALID_NTREE_IDX;
418  m_ElementCount = 0;
419 }

435 // Links a node after a particular node
436 //-----------------------------------------------------------------------------
437 template <class T, class I>
438 void CUtlNTree<T,I>::LinkChildAfter( I parent, I after, I elem )
439 {
440  Assert( IsInTree(elem) );
441 
442  // Unlink it if it's in the list at the moment
443  Unlink(elem);
444 
445  Node_t& newElem = InternalNode(elem);
446  newElem.m_Parent = parent;
447  newElem.m_PrevSibling = after;
448  if ( after != INVALID_NTREE_IDX )
449  {
450  Node_t& prevSiblingNode = InternalNode( after );
451  newElem.m_NextSibling = prevSiblingNode.m_NextSibling;
452  prevSiblingNode.m_NextSibling = elem;
453  }
454  else
455  {
456  if ( parent != INVALID_NTREE_IDX )
457  {
458  Node_t& parentNode = InternalNode( parent );
459  newElem.m_NextSibling = parentNode.m_FirstChild;
460  parentNode.m_FirstChild = elem;
461  }
462  else
463  {
464  newElem.m_NextSibling = m_Root;
465  if ( m_Root != INVALID_NTREE_IDX )
466  {
467  Node_t& rootNode = InternalNode( m_Root );
468  rootNode.m_PrevSibling = elem;
469  }
470  m_Root = elem;
471  }
472  }
473 
474  if ( newElem.m_NextSibling != INVALID_NTREE_IDX )
475  {
476  Node_t& nextSiblingNode = InternalNode( newElem.m_NextSibling );
477  nextSiblingNode.m_PrevSibling = elem;
478  }
479 }

482 //-----------------------------------------------------------------------------
483 // Links a node before a particular node
484 //-----------------------------------------------------------------------------
485 template <class T, class I>
486 void CUtlNTree<T,I>::LinkChildBefore( I parent, I before, I elem )
487 {
488  Assert( IsValidIndex(elem) );
489 
490  if ( before != INVALID_NTREE_IDX )
491  {
492  LinkChildAfter( parent, InternalNode( before ).m_PrevSibling, elem );
493  return;
494  }
495 
496  // NOTE: I made the choice to do an O(n) operation here
497  // instead of store more data per node (LastChild).
498  // This might not be the right choice. Revisit if we get perf problems.
499  I after;
500  if ( parent != INVALID_NTREE_IDX )
501  {
502  after = InternalNode( parent ).m_FirstChild;
503  }
504  else
505  {
506  after = m_Root;
507  }
508 
509  if ( after == INVALID_NTREE_IDX )
510  {
511  LinkChildAfter( parent, after, elem );
512  return;
513  }
514 
515  I next = InternalNode( after ).m_NextSibling;
516  while ( next != InvalidIndex() )
517  {
518  after = next;
519  next = InternalNode( next ).m_NextSibling;
520  }
521 
522  LinkChildAfter( parent, after, elem );
523 }

526 //-----------------------------------------------------------------------------
527 // Unlinks a node from the tree
528 //-----------------------------------------------------------------------------
529 template <class T, class I>
530 void CUtlNTree<T,I>::Unlink( I elem )
531 {
532  Assert( IsInTree(elem) );
533 
534  Node_t *pOldNode = &InternalNode( elem );
535 
536  // If we're the first guy, reset the head
537  // otherwise, make our previous node's next pointer = our next
538  if ( pOldNode->m_PrevSibling != INVALID_NTREE_IDX )
539  {
540  InternalNode( pOldNode->m_PrevSibling ).m_NextSibling = pOldNode->m_NextSibling;
541  }
542  else
543  {
544  if ( pOldNode->m_Parent != INVALID_NTREE_IDX )
545  {
546  InternalNode( pOldNode->m_Parent ).m_FirstChild = pOldNode->m_NextSibling;
547  }
548  else if ( m_Root == elem )
549  {
550  m_Root = pOldNode->m_NextSibling;
551  }
552  }
553 
554  // If we're the last guy, reset the tail
555  // otherwise, make our next node's prev pointer = our prev
556  if ( pOldNode->m_NextSibling != INVALID_NTREE_IDX )
557  {
558  InternalNode( pOldNode->m_NextSibling ).m_PrevSibling = pOldNode->m_PrevSibling;
559  }
560 
561  // Unlink everything except children
562  pOldNode->m_Parent = pOldNode->m_PrevSibling = pOldNode->m_NextSibling = INVALID_NTREE_IDX;
563 }
