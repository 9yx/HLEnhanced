96 inline void *MemAlloc_AllocAligned( size_t size, size_t align )
97 {
98  unsigned char *pAlloc, *pResult;
99 
100  if (!IsPowerOfTwo(align))
101  return NULL;
102 
103  align = (align > sizeof(void *) ? align : sizeof(void *)) - 1;
104 
105  if ( (pAlloc = (unsigned char*)g_pMemAlloc->Alloc( sizeof(void *) + align + size ) ) == (unsigned char*)NULL)
106  return NULL;
107 
108  pResult = (unsigned char*)( (unsigned)(pAlloc + sizeof(void *) + align ) & ~align );
109  ((unsigned char**)(pResult))[-1] = pAlloc;
110 
111  return (void *)pResult;
112 }

132 inline void MemAlloc_FreeAligned( void *pMemBlock )
133 {
134  unsigned *pAlloc;
135 
136  if ( pMemBlock == NULL )
137  return;
138 
139  pAlloc = (unsigned *)pMemBlock;
140 
141  // pAlloc points to the pointer to starting of the memory block
142  pAlloc = (unsigned *)(((unsigned)pAlloc & ~( sizeof(void *) - 1 ) ) - sizeof(void *));
