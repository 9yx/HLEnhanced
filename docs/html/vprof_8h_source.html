493 #ifdef VPROF_VTUNE_GROUP
494 inline void CVProfile::PushGroup( int nGroupID )
495 {
496  // There is always at least one item on the stack since we force
497  // the first element to be VPROF_BUDGETGROUP_OTHER_UNACCOUNTED.
498  Assert( m_GroupIDStackDepth > 0 );
499  Assert( m_GroupIDStackDepth < MAX_GROUP_STACK_DEPTH );
500  m_GroupIDStack[m_GroupIDStackDepth] = nGroupID;
501  m_GroupIDStackDepth++;
502  if( m_GroupIDStack[m_GroupIDStackDepth-2] != nGroupID &&
503  VTuneGroupEnabled() &&
504  nGroupID == VTuneGroupID() )
505  {
506  vtune( true );
507  }
508 }
509 #endif // VPROF_VTUNE_GROUP
510 
511 #ifdef VPROF_VTUNE_GROUP
512 inline void CVProfile::PopGroup( void )
513 {
514  m_GroupIDStackDepth--;
515  // There is always at least one item on the stack since we force
516  // the first element to be VPROF_BUDGETGROUP_OTHER_UNACCOUNTED.
517  Assert( m_GroupIDStackDepth > 0 );
518  if( m_GroupIDStack[m_GroupIDStackDepth] != m_GroupIDStack[m_GroupIDStackDepth+1] &&
519  VTuneGroupEnabled() &&
520  m_GroupIDStack[m_GroupIDStackDepth+1] == VTuneGroupID() )
521  {
522  vtune( false );
523  }
524 }
525 #endif // VPROF_VTUNE_GROUP
