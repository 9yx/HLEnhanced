291 inline void bf_write::WriteUBitLong( unsigned int curData, int numbits, bool bCheckRange )
292 {
293 #ifdef _DEBUG
294  // Make sure it doesn't overflow.
295  if ( bCheckRange && numbits < 32 )
296  {
297  if ( curData >= (unsigned long)(1 << numbits) )
298  {
299  CallErrorHandler( BITBUFERROR_VALUE_OUT_OF_RANGE, GetDebugName() );
300  }
301  }
302  Assert( numbits >= 0 && numbits <= 32 );
303 #endif
304 
305  extern unsigned long g_BitWriteMasks[32][33];
306 
307  // Bounds checking..
308  if((m_iCurBit+numbits) > m_nDataBits)
309  {
310  m_iCurBit = m_nDataBits;
311  SetOverflowFlag();
312  CallErrorHandler( BITBUFERROR_BUFFER_OVERRUN, GetDebugName() );
313  return;
314  }
315 
316  int nBitsLeft = numbits;
317  int iCurBit = m_iCurBit;
318 
319  // Mask in a dword.
320  unsigned int iDWord = iCurBit >> 5;
321  Assert( (iDWord*4 + sizeof(long)) <= (unsigned int)m_nDataBytes );
322 
323  unsigned long iCurBitMasked = iCurBit & 31;
324  ((unsigned long*)m_pData)[iDWord] &= g_BitWriteMasks[iCurBitMasked][nBitsLeft];
325  ((unsigned long*)m_pData)[iDWord] |= curData << iCurBitMasked;
326 
327  // Did it span a dword?
328  int nBitsWritten = 32 - iCurBitMasked;
329  if(nBitsWritten < nBitsLeft)
330  {
331  nBitsLeft -= nBitsWritten;
332  iCurBit += nBitsWritten;
333  curData >>= nBitsWritten;
334 
335  unsigned long iCurBitMasked = iCurBit & 31;
336  ((unsigned long*)m_pData)[iDWord+1] &= g_BitWriteMasks[iCurBitMasked][nBitsLeft];
337  ((unsigned long*)m_pData)[iDWord+1] |= curData << iCurBitMasked;
338  }
339 
340  m_iCurBit += numbits;
341 }

591 inline unsigned int bf_read::ReadUBitLong( int numbits )
592 {
593  extern unsigned long g_ExtraMasks[32];
594 
595  if ( (m_iCurBit+numbits) > m_nDataBits )
596  {
597  m_iCurBit = m_nDataBits;
598  SetOverflowFlag();
599  return 0;
600  }
601 
602  Assert( numbits > 0 && numbits <= 32 );
603 
604  // Read the current dword.
605  int idword1 = m_iCurBit >> 5;
606  unsigned int dword1 = ((unsigned int*)m_pData)[idword1];
607  dword1 >>= (m_iCurBit & 31); // Get the bits we're interested in.
608 
609  m_iCurBit += numbits;
610  unsigned int ret = dword1;
611 
612  // Does it span this dword?
613  if ( (m_iCurBit-1) >> 5 == idword1 )
614  {
615  if(numbits != 32)
616  ret &= g_ExtraMasks[numbits];
617  }
618  else
619  {
620  int nExtraBits = m_iCurBit & 31;
621  unsigned int dword2 = ((unsigned int*)m_pData)[idword1+1] & g_ExtraMasks[nExtraBits];
622 
623  // No need to mask since we hit the end of the dword.
624  // Shift the second dword's part into the high bits.
625  ret |= (dword2 << (numbits - nExtraBits));
626  }
627 
628  return ret;
