85 // NJS: Jcalg helper functions:
86 // Decompress a chunk of a buffer compressed via xCompress.
87 // What you want to do here is read a 2 meg chunk from the file and pass it to this function with an nMaxOutput that is at least 4x nInputLength
88 // the return value will be the number of bytes decompressed into the output buffer, or 0xFFFFFFFF if the output buffer overflowed.
89 inline unsigned JCALG1_Decompress_Formatted_Buffer( unsigned nInputLength, void* pInput, unsigned nMaxOutput, void* pOutput )
90 {
91  unsigned outputBufferLength = 0;
92  for( char* pInputPointer = (char*)pInput; pInputPointer < (char*)pInput+nInputLength; )
93  {
94  // Get the size of the next chunk:
95  unsigned short bytesThisBlock = *(unsigned short*)pInputPointer;
96 
97  // A zero sized byte block indicates I've hit the end of the buffer:
98  if( !bytesThisBlock )
99  break;
100 
101  pInputPointer += sizeof(unsigned short);
102 
103  // Is this block compressed?
104  if( bytesThisBlock & 0x8000 )
105  {
106  bytesThisBlock &= ~0x8000;
107 
108  // No it isn't just copy it:
109  memcpy((char*)pOutput + outputBufferLength, pInputPointer, bytesThisBlock );
110 
111  outputBufferLength += bytesThisBlock;
112  pInputPointer += bytesThisBlock;
113  }
114  else
115  {
116  unsigned bytesDecompressed = JCALG1_Decompress_Fast( pInputPointer, (char*)pOutput + outputBufferLength);
117 
118  outputBufferLength += bytesDecompressed;
119  pInputPointer += bytesThisBlock;
120  }
121 
122  if( outputBufferLength > nMaxOutput )
123  {
124  return 0xFFFFFFFF;
125  }
126  }
127 
128  return outputBufferLength;
129 }
