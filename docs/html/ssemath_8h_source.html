53 static inline __m128 MMReciprocal(__m128 x)
54 {
55  __m128 ret=_mm_rcp_ps(x);
56  // newton iteration is Y(n+1)=2*Y(N)-x*Y(n)^2
57  ret=_mm_sub_ps(_mm_add_ps(ret,ret),_mm_mul_ps(x,_mm_mul_ps(ret,ret)));
58  return ret;

290  inline void VectorNormalize(void)
291  {
292  static __m128 FourHalves={0.5,0.5,0.5,0.5};
293  static __m128 FourThrees={3.,3.,3.,3.};
294  __m128 mag_sq=(*this)*(*this); // length^2
295  __m128 guess=_mm_rsqrt_ps(mag_sq);
296  // newton iteration for 1/sqrt(x) : y(n+1)=1/2 (y(n)*(3-x*y(n)^2));
297  guess=_mm_mul_ps(guess,_mm_sub_ps(FourThrees,_mm_mul_ps(mag_sq,_mm_mul_ps(guess,guess))));
298  guess=_mm_mul_ps(Four_PointFives,guess);
299  (*this) *= guess;
300  }

381 // PowSSE - raise an sse register to a power. This is analogous to the C pow() function, with some
382 // restictions: fractional exponents are only handled with 2 bits of precision. Basically,
383 // fractions of 0,.25,.5, and .75 are handled. PowSSE(x,.30) will be the same as PowSSE(x,.25).
384 // negative and fractional powers are handled by the SSE reciprocal and square root approximation
385 // instructions and so are not especially accurate ----Note that this routine does not raise
386 // numeric exceptions because it uses SSE--- This routine is O(log2(exponent)).
387 inline __m128 PowSSE(__m128 x, float exponent)
388 {
389  return PowSSE_FixedPoint_Exponent(x,(int) (4.0*exponent));
390 }
