48 // NOTE: The following methods must be implemented in your application
49 // although they can be empty implementations if you like...
50 //-----------------------------------------------------------------------------
51 abstract_class IAppSystemGroup
52 {
53 public:
54  // An installed application creation function, you should tell the group
55  // the DLLs and the singleton interfaces you want to instantiate.
56  // Return false if there's any problems and the app will abort
57  virtual bool Create( ) = 0;
58 
59  // Allow the application to do some work after AppSystems are connected but
60  // they are all Initialized.
61  // Return false if there's any problems and the app will abort
62  virtual bool PreInit() = 0;
63 
64  // Main loop implemented by the application
65  virtual int Main( ) = 0;
66 
67  // Allow the application to do some work after all AppSystems are shut down
68  virtual void PostShutdown() = 0;
69 
70  // Call an installed application destroy function, occurring after all modules
71  // are unloaded
72  virtual void Destroy() = 0;
73 };
