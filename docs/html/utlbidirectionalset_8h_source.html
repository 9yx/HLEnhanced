20 //-----------------------------------------------------------------------------
21 // Templatized helper class to deal with the kinds of things that spatial
22 // partition code always seems to have; buckets with lists of lots of elements
23 // and elements that live in lots of buckets. This makes it really quick to
24 // add and remove elements, and to iterate over all elements in a bucket.
25 //
26 // For this to work, you must initialize the set with two functions one that
27 // maps from bucket to the index of the first element in that bucket, and one
28 // that maps from element to the index of the first bucket that element lies in.
29 // The set will completely manage the index, it's just expected that those
30 // indices will be stored outside the set.
31 //-----------------------------------------------------------------------------
32 template< class CBucketHandle, class CElementHandle, class I >
33 class CBidirectionalSet

35 public:
36  // Install methods to get at the first bucket given a element
37  // and vice versa...

125 //-----------------------------------------------------------------------------
126 // Adds an element to the bucket
127 //-----------------------------------------------------------------------------
128 template< class CBucketHandle, class CElementHandle, class I >
129 void CBidirectionalSet<CBucketHandle,CElementHandle,I>::AddElementToBucket( CBucketHandle bucket, CElementHandle element )
130 {
131  Assert( m_FirstBucket && m_FirstElement );
132 
133 #ifdef _DEBUG
134  // Make sure that this element doesn't already exist in the list of elements in the bucket
135  I elementInBucket = m_FirstElement( bucket );
136  while( elementInBucket != m_ElementsInBucket.InvalidIndex() )
137  {
138  // If you hit an Assert here, fix the calling code. It's too expensive to ensure
139  // that each item only shows up once here. Hopefully you can do something better
140  // outside of here.
141  Assert( m_ElementsInBucket[elementInBucket].m_Element != element );
142  elementInBucket = m_ElementsInBucket.Next( elementInBucket );
143  }
144 #endif
145 
146 #ifdef _DEBUG
147  // Make sure that this bucket doesn't already exist in the element's list of buckets.
148  I bucketInElement = m_FirstBucket( element );
149  while( bucketInElement != m_BucketsUsedByElement.InvalidIndex() )
150  {
151  // If you hit an Assert here, fix the calling code. It's too expensive to ensure
152  // that each item only shows up once here. Hopefully you can do something better
153  // outside of here.
154  Assert( m_BucketsUsedByElement[bucketInElement].m_Bucket != bucket );
155  bucketInElement = m_BucketsUsedByElement.Next( bucketInElement );
156  }
157 #endif
158 
159  // Allocate new element + bucket entries
160  I idx = m_ElementsInBucket.Alloc(true);
161  I list = m_BucketsUsedByElement.Alloc( true );
162 
163  // Store off the element data
164  m_ElementsInBucket[idx].m_Element = element;
165  m_ElementsInBucket[idx].m_BucketListIndex = list;
166 
167  // Here's the bucket data
168  m_BucketsUsedByElement[list].m_Bucket = bucket;
169  m_BucketsUsedByElement[list].m_ElementListIndex = idx;
170 
171  // Insert the element into the list of elements in the bucket
172  I& firstElementInBucket = m_FirstElement( bucket );
173  if ( firstElementInBucket != m_ElementsInBucket.InvalidIndex() )
174  m_ElementsInBucket.LinkBefore( firstElementInBucket, idx );
175  firstElementInBucket = idx;
176 
177  // Insert the bucket into the element's list of buckets
178  I& firstBucketInElement = m_FirstBucket( element );
179  if ( firstBucketInElement != m_BucketsUsedByElement.InvalidIndex() )
180  m_BucketsUsedByElement.LinkBefore( firstBucketInElement, list );
181  firstBucketInElement = list;
182 }

188 template< class CBucketHandle, class CElementHandle, class I >
189 bool CBidirectionalSet<CBucketHandle,CElementHandle,I>::IsElementInBucket( CBucketHandle bucket, CElementHandle element )
190 {
191  // Search through all elements in this bucket to see if element is in there.
192  I elementInBucket = m_FirstElement( bucket );
193  while( elementInBucket != m_ElementsInBucket.InvalidIndex() )
194  {
195  if( m_ElementsInBucket[elementInBucket].m_Element == element )
196  {
197  return true;
198  }
199  elementInBucket = m_ElementsInBucket.Next( elementInBucket );
200  }
201  return false;
202 }

216 //-----------------------------------------------------------------------------
217 // Removes an element from all buckets
218 //-----------------------------------------------------------------------------
219 template< class CBucketHandle, class CElementHandle, class I >
220 void CBidirectionalSet<CBucketHandle,CElementHandle,I>::RemoveElement( CElementHandle element )
221 {
222  Assert( m_FirstBucket && m_FirstElement );
223 
224  // Iterate over the list of all buckets the element is in
225  I i = m_FirstBucket( element );
226  while (i != m_BucketsUsedByElement.InvalidIndex())
227  {
228  CBucketHandle bucket = m_BucketsUsedByElement[i].m_Bucket;
229  I elementListIndex = m_BucketsUsedByElement[i].m_ElementListIndex;
230 
231  // Unhook the element from the bucket's list of elements
232  if (elementListIndex == m_FirstElement(bucket))
233  m_FirstElement(bucket) = m_ElementsInBucket.Next(elementListIndex);
234  m_ElementsInBucket.Free(elementListIndex);
235 
236  unsigned short prevNode = i;
237  i = m_BucketsUsedByElement.Next(i);
238  m_BucketsUsedByElement.Free(prevNode);
239  }
240 
241  // Mark the list as empty
242  m_FirstBucket( element ) = m_BucketsUsedByElement.InvalidIndex();
243 }

245 //-----------------------------------------------------------------------------
246 // Removes a bucket from all elements
247 //-----------------------------------------------------------------------------
248 template< class CBucketHandle, class CElementHandle, class I >
249 void CBidirectionalSet<CBucketHandle,CElementHandle,I>::RemoveBucket( CBucketHandle bucket )
250 {
251  // Iterate over the list of all elements in the bucket
252  I i = m_FirstElement( bucket );
253  while (i != m_ElementsInBucket.InvalidIndex())
254  {
255  CElementHandle element = m_ElementsInBucket[i].m_Element;
256  I bucketListIndex = m_ElementsInBucket[i].m_BucketListIndex;
257 
258  // Unhook the bucket from the element's list of buckets
259  if (bucketListIndex == m_FirstBucket(element))
260  m_FirstBucket(element) = m_BucketsUsedByElement.Next(bucketListIndex);
261  m_BucketsUsedByElement.Free(bucketListIndex);
262 
263  // Remove the list element
264  unsigned short prevNode = i;
265  i = m_ElementsInBucket.Next(i);
266  m_ElementsInBucket.Free(prevNode);
267  }
268 
269  // Mark the bucket list as empty
270  m_FirstElement( bucket ) = m_ElementsInBucket.InvalidIndex();
271 }
