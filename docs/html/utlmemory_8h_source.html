399 //-----------------------------------------------------------------------------
400 // Grows the memory
401 //-----------------------------------------------------------------------------
402 inline int UtlMemory_CalcNewAllocationCount( int nAllocationCount, int nGrowSize, int nNewSize, int nBytesItem )
403 {
404  if ( nGrowSize )
405  {
406  nAllocationCount = ((1 + ((nNewSize - 1) / nGrowSize)) * nGrowSize);
407  }
408  else
409  {
410  if ( !nAllocationCount )
411  {
412  // Compute an allocation which is at least as big as a cache line...
413  nAllocationCount = (31 + nBytesItem) / nBytesItem;
414  }
415 
416  while (nAllocationCount < nNewSize)
417  {
418 #ifndef _XBOX
419  nAllocationCount *= 2;
420 #else
421  int nNewAllocationCount = ( nAllocationCount * 9) / 8; // 12.5 %
422  if ( nNewAllocationCount > nAllocationCount )
423  nAllocationCount = nNewAllocationCount;
424  else
425  nAllocationCount *= 2;
426 #endif
427  }
428  }
429 
430  return nAllocationCount;
431 }

433 template< class T >
434 void CUtlMemory<T>::Grow( int num )
435 {
436  Assert( num > 0 );
437 
438  if ( IsExternallyAllocated() )
439  {
440  // Can't grow a buffer whose memory was externally allocated
441  Assert(0);
442  return;
443  }
444 
445  // Make sure we have at least numallocated + num allocations.
446  // Use the grow rules specified for this memory (in m_nGrowSize)
447  int nAllocationRequested = m_nAllocationCount + num;
448 
449  UTLMEMORY_TRACK_FREE();
450 
451  m_nAllocationCount = UtlMemory_CalcNewAllocationCount( m_nAllocationCount, m_nGrowSize, nAllocationRequested, sizeof(T) );
452 
453  UTLMEMORY_TRACK_ALLOC();
454 
455  if (m_pMemory)
456  {
457  MEM_ALLOC_CREDIT_CLASS();
458  m_pMemory = (T*)realloc( m_pMemory, m_nAllocationCount * sizeof(T) );
459  Assert( m_pMemory );
460  }
461  else
462  {
463  MEM_ALLOC_CREDIT_CLASS();
464  m_pMemory = (T*)malloc( m_nAllocationCount * sizeof(T) );
465  Assert( m_pMemory );
466  }

523 template< class T >
524 void CUtlMemory<T>::Purge( int numElements )
525 {
526  Assert( numElements >= 0 );
527 
528  if( numElements > m_nAllocationCount )
529  {
530  // Ensure this isn't a grow request in disguise.
531  Assert( numElements <= m_nAllocationCount );
532  return;
533  }
534 
535  // If we have zero elements, simply do a purge:
536  if( numElements == 0 )
537  {
538  Purge();
539  return;
540  }
541 
542  if ( IsExternallyAllocated() )
543  {
544  // Can't shrink a buffer whose memory was externally allocated, fail silently like purge
545  return;
546  }
547 
548  // If the number of elements is the same as the allocation count, we are done.
549  if( numElements == m_nAllocationCount )
550  {
551  return;
552  }
553 
554 
555  if( !m_pMemory )
556  {
557  // Allocation count is non zero, but memory is null.
558  Assert( m_pMemory );
559  return;
560  }
561 
562  UTLMEMORY_TRACK_FREE();
563 
564  m_nAllocationCount = numElements;
565 
566  UTLMEMORY_TRACK_ALLOC();
567 
568  // Allocation count > 0, shrink it down.
569  MEM_ALLOC_CREDIT_CLASS();
570  m_pMemory = (T*)realloc( m_pMemory, m_nAllocationCount * sizeof(T) );
571 }
